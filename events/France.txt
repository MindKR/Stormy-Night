add_namespace = france

#############
### SYSTEM ###
#############

# Many Marxists
country_event = {
	id = france.1

	hidden = yes

	trigger = {
		has_country_leader = { name = "Commune Council (Marxist)" }
		tag = FRA
	}

	immediate = {
		hidden_effect = {
			set_country_flag = FRA_marxist
			clr_country_flag = FRA_leninist
			clr_country_flag = FRA_stalinist
			clr_country_flag = FRA_anarchist 
		}
	}
}
