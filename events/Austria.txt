# Ferdinand dies
country_event = {
	id = austria.2
	title = austria.2.t
	desc = austria.2.d
	picture = GFX_report_event_generic_funeral

	fire_only_once = yes

	is_triggered_only = yes
	
	trigger = {
		original_tag = AUS
		NOT = { has_country_flag = ferdinand_dead }
		has_country_leader = { character = AUS_Franz_Ferdinand }
	}

	option = {
		name = austria.2.a
		Ferdinand_dies = yes
	}
}
